<body>
    <h2>نظام الإشعارات المباشر</h2>
    <div id="messages"></div>

    <script type="text/javascript">
        var socket = new SockJS('https://localhost:8443/ws');
        var stompClient = Stomp.over(socket);

        stompClient.connect({}, function (frame) {
            console.log('Connected: ' + frame);
            
            stompClient.subscribe('/topic/updates', function (notification) {
                // لاحظ هنا: قمنا بتغيير القناة إلى /topic/updates لتطابق الكود في NotificationService
                showMessage(notification.body);
            });
        });

        function showMessage(message) {
            var response = document.getElementById('messages');
            var p = document.createElement('p');
            p.style.wordWrap = 'break-word';
            p.appendChild(document.createTextNode(message));
            response.appendChild(p);
        }
    </script>
